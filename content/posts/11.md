$$ 
\begin{aligned}
v_\pi(s) 
&= \mathbb{E}_\pi[G_t | S_t = s]
\end{aligned}
$$ 

$$ 
\begin{aligned}
v_\pi(s) 
&= \sum_a\pi(a|s) \sum_{s^\prime, r}p(s^\prime, r | s, a)[r + \gamma v_\pi(s^\prime)], \text{ for all } s \in S
\end{aligned}
$$
